Using a for loop to print Array elements

You can use a 'for' loop to print out each index value a specific number of times This is not possible with forEach() since that only prints the exact amount of indexes in the Array.

This returns:

  • Pineapple
  • Lemon
  • Apple
  • Orange
  • Peach
var fruits = ["Pineapple", "Lemon", "Apple", "Orange", "Peach"]

for(var i=0; i < fruits.length; i++) {
console.log(fruits[i])
}

Using a forEach() function

You can also use a 'for' loop to print out each index value.

This returns the elments in an HTML unordered list. For example.

<ul><li>Banana</li><li>Orange</li><li>Apple</li><li>Mango</li></ul>

var fruits, text;
fruits = ["Banana", "Orange", "Apple", "Mango"];

text = "<ul>";
fruits.forEach(myFunction);
text += "</ul>";

function myFunction(value) {
  text += "<li>" + value + "</li>";
}

console.log(text);

For this reason, it makes sense to use getElementById to add this information to a page.

For example, this prints the unordered list to the HTML page.

<p id="demo">

var fruits, text;
fruits = ["Banana", "Orange", "Apple", "Mango"];

text = "<ul>";
fruits.forEach(myFunction);
text += "</ul>";
document.getElementById("demo").innerHTML = text;

function myFunction(value) {
  text += "<li>" + value + "</li>";
}
