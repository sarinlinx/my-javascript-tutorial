Sorting Arrays with sort()

Alpha Sorting

sort() puts alpha strings in alpa order.

This prints: Apple,Banana,Mango

var fruits = ["Banana", "Apple", "Mango"];
console.log(fruits.sort());

Numeric sorting

This outputs: Array(3) [ 1, 15, 2 ]

Notice something strange? The order is 1, 15, 2.

Thatâ€™s because sort() converts everything to string and uses lexicographical order by default.

var arr = [ 1, 2, 15 ]
arr.sort()
console.log(arr);

Sorting in correct order using a comparison function

RE-read this to fully understand

To make it smarter, we need to pass in the custom comparison function.

It should accept two arguments and return 1, 0 or -1.

The purpose of the compare function is to define an alternative sort order.

The compare function should return a negative, zero, or positive value, depending on the arguments

Syntax

Example of a comparison function

function(a, b){return a-b}

Example (increasing order)

This sorts in increasing order.

var points = [40, 100, 1, 5, 25, 10];

function myFunction() {
    points.sort(function(a, b)
    {return a - b}
  );
  console.log(points);
}

myFunction();

Example (decreasing order)

This sorts in increasing order.

var points = [40, 100, 1, 5, 25, 10];

function myFunction() {
    points.sort(function(a, b)
    {return b - a}
  );
  console.log(points);
}

myFunction();

Another example

This outputs: Array(3) [ 1, 2, 15 ]

function compare(a, b) {
  if (a > b) return 1
  else if (a < b) return -1
  else return 0
}

var arr = [ 1, 2, 15 ]
arr.sort(compare)
console.log(arr);

Random sorting

Use a compare function with Math.random()

var points = [40, 100, 1, 5, 25, 10];
points.sort(function(a, b) {
  return 0.5 - Math.random()

  }
);

console.log(points);

Find the Highest (or Lowest) Array Value

Sorting

Sorting a whole array is a very inefficient method if you only want to find the highest (or lowest) value.

There are no built-in functions for finding the max or min value in an array.

However, after you have sorted an array, you can use the index to obtain the highest and lowest values.

var points = [40, 100, 1, 5, 25, 10];
points.sort(function(a, b){
  return a-b
});

console.log(points[0]);

Using Math.max()

Find highest value

Use Math.max.apply to find the highest number in an array

var points = [40, 100, 1, 5, 25, 10];

function myArrayMax(arr) {
    return Math.max.apply(null, arr);
}

console.log(myArrayMax(points));
