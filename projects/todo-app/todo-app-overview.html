Todo List

Only return the Todo List items that are not completed.

These are marked as false.

const todos = [{
  text: 'Order cat food',
  completed: false
}, {
  text: 'Clean kitchen',
  completed: true
}, {
  text: 'Buy food',
  completed: true
}, {
  text: 'Do work',
  completed: false
}, {
  text: 'Exercise',
  completed: true
}]



const getThingsToDo = function(todosArray) {
return todosArray.filter(function(todoz) {
    return todoz.completed === false
    // you could also do it this way
    // return !todo.completed
})
}


console.log(getThingsToDo(todos))
Output

This example outputs:

[
{ text: 'Order cat food', completed: false },
{ text: 'Do work', completed: false }
]

V1

This removes all paragraphs with the word 'the'.

View the example here:

  • todo-app (v1)
<h1>Todos</h1>
<p>Clean the kitchen</p>
<p>Book flights</p>
<p>Reserach museums</p>
<p>Walk the dog</p>
<p>Finish this course!</p>

// target all paragraphs
const paragraphs = document.querySelectorAll('p')
              
paragraphs.forEach(function(paragraph) {
  // used 'textContent.includes()' to check if the paragraph's
  // text includes the word 'the'
  if (paragraph.textContent.includes('the')) {
    // remove those paragraphs
      paragraph.remove()
  }
})
Output

This example outputs:

Todos
Book flights
Reserach museums
Finish this course!

V2

This first prints how many Todos are left to complete.

It then prints ALL todos.

View the example here:

  • todo-app (v2)
                  <h1>h1Todos</h1>
                  
  const todos = [{
  text: 'Order cat food',
  completed: false
}, {
  text: 'Clean kitchen',
  completed: true
}, {
  text: 'Buy food',
  completed: true
}, {
  text: 'Do work',
  completed: false
}, {
  text: 'Exercise',
  completed: true
}]

// store items NOT complete in const
const incompleteTodos = todos.filter(function(todo) {
  return !todo.completed
})

// create an h2 element
const summary = document.createElement('h2')

// add text content
summary.textContent = `You have ${incompleteTodos.length} todos left`

// add element to the bottom of the page
document.querySelector('body').appendChild(summary)

// loop over Array of Objects to add ALL to the page
todos.forEach(function(todo) {
  const p = document.createElement('p')
  p.textContent = todo.text
  document.querySelector('body').appendChild(p)
})
Output

This example outputs:

Todos
You have 2 todos left                  
Order cat food                  
Clean kitchen                  
Buy food                  
Do work
