const notes = [{
    title: 'my next trip',
    body: 'I would like to go to Spain'
}, {
    title: 'Habbits to work on',
    body: 'Exercise. Eating a bit better.'
}, {
    title: 'Office modification',
    body: 'Get a new seat'
}]

//Create Object to keep track of latest data
//This searchText property changes as search input changes
const filters = {
    searchText: ''
}

//Function to filter Array
//This takes in 2 params, the Array 'notes' and the Object 'filters'
const renderNotes = function (notes, filters) {
   //filter using the filter() which creates a
   //new Array named filteredTodos
   //x is the full todos Array
   const filteredNotes = notes.filter(function (x) {
   //use includes() to check if it contains your input text value
   //stored in the property 'searchText' 
   return x.title.toLowerCase().includes(filters.searchText.toLowerCase())
    })

    document.querySelector('#notes').innerHTML = ''

    filteredNotes.forEach(function (note) {
        const noteEl = document.createElement('p')
        noteEl.textContent = note.title
        document.querySelector('#notes').appendChild(noteEl)
    })
}

//call initial rendering of Array elements
renderNotes(notes, filters)

document.querySelector('#create-note').addEventListener('click', function (e) {
    //when button clicked, change text
    e.target.textContent = 'The button was clicked'
})

document.querySelector('#search-text').addEventListener('input', function (e) {
    //Change filters Object property of searchText to input text
    filters.searchText = e.target.value
    //Run function to update Object
    renderNotes(notes, filters)
})

document.querySelector('#name-form').addEventListener('submit', function (e) {
    //cancel default submit
    e.preventDefault()
    console.log(e.target.elements.firstName.value)
    //access form data
    //the elements property allows you to access all the fields in the form
    //this specifically grabs the firstName element using the value property
    //it is then set to an empty string. This is so the field is cleared when
    //submit button is clicked
    e.target.elements.firstName.value = ''
})
